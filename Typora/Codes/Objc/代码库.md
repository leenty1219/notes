## 几种安全的删除数组元素的方式
```objc
for (NSInteger i = self.selectedAssets.count - 1; i >= 0; i--) {
    SMUIPHAsset *element = self.selectedAssets[i];

        [self.selectedAssets removeObjectAtIndex:i];

    }

// 第一种采用倒叙的方式，因为从后往前便利不会导致下标越界的问题

NSUInteger index = [self.selectedAssets indexOfObjectPassingTest:^BOOL(SMUIPHAsset *element, NSUInteger idx, BOOL *stop) {

[self.selectedAssets filterUsingPredicate:[NSPredicate predicateWithBlock:^BOOL(SMUIPHAsset *evaluatedObject, NSDictionary *bindings) {
```

```objc
    self.monitorQueue = dispatch_queue_create("network.monitor", DISPATCH_QUEUE_SERIAL);
   // 3. 设置回调队列
    nw_path_monitor_set_queue(self.monitor, self.monitorQueue);

    // 4. 设置网络状态变化的回调 Block
    nw_path_monitor_set_update_handler(self.monitor, ^(nw_path_t  _Nonnull path) {
        nw_path_status_t status = nw_path_get_status(path); // 获取当前的连接状态
        nw_path_unsatisfied_reason_t reason = nw_path_get_unsatisfied_reason(nw_path_t path); // 获取当前断网原因
      	nw_path_has_ipv4(path);

        nw_link_quality_t quality = nw_path_get_link_quality(path); // 获得当前的网络连接质量
        nw_path_enumerate_interfaces(path, ^bool(nw_interface_t  _Nonnull interface) { // 遍历当前的interface集合

                  nw_interface_type_t t = nw_interface_get_type(interface); // 获取当前interface类型

              }

         });
      
    });

    // 5. 开始监听s
    nw_path_monitor_start(self.monitor);
}
```
