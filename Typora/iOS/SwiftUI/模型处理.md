## 1.简单实例

`@StateObject` 会把对象存到 SwiftUI 的状态系统。

`ObservableObject`表示：这个对象变化时可以通知 UI。本质上是*是引用类型观察系统*。`class VM: ObservableObject`

`@published`：name 改变 -> 发送 objectWillChange -> SwiftUI 刷新依赖 View

更推荐的结构：

> `struct`  ：**Model**

>  `class` ：**ViewModel**

> `struct`： **View**

 

`@StateObject`：当前 View 创建并拥有 VM

`@ObservedObject`：VM 是外面传进来的，当前 View 不拥有它



```swift
/// ViewModl
class UserVM: ObservableObject {
    @Published var name = "Tom"
}

/// View
struct UserView: View {

    @StateObject var vm = UserVM() //当前 View 创建并拥有 VM
  	@ObservedObject var vm = UserVM //VM 是外面传进来的，当前 View 不拥有它。所以不应该初始化

    var body: some View {
        Text(vm.name)
    }
}
```

### 1.1 Model

SwiftUI 强烈推荐Model使用`struct`，**SwiftUI 喜欢值语义**

有点：

* 更安全
* Diff 更容易
* 不会引用共享
* 更适合状态快照

### 1.2 ViewModel

```swift
/// 一个简单的用户登录模型，⚠️这里有个问题，模型全部可观察，导致会出现大量刷新
class ProfileViewModel: ObservableObject {

    @Published var avatar = ""

    @Published var nickname = ""

    @Published var loading = false

    @Published var buttonEnable = true

    @Published var inputText = ""

/// 这里不要直接依赖ProfileViewModel，要更小粒度的使用

        }
    }

    nickname: vm.user.nickname
```

### 1.3 可编辑ViewModel

```swift
// 模型属性尽量不可变，使用let
// Model 是服务器数据快照，不是UI编辑状态
struct User {
    let id: String
    let avatar: String
    let nickname: String
    let vipLevel: Int
    let phone: String
    let email: String
    ...

// ViewModel
class ProfileVM: ObservableObject {
    private var user: User
    @Published var avatar: String
    @Published var nickname: String
    init(user: User) {
        self.user = user
        self.avatar = user.avatar
        self.nickname = user.nickname

//View
struct ProfileView: View {

    var body: some View {
        VStack {
            Image(vm.avatar)
            TextField(
                "nickname",
                text: $vm.nickname
            )
        }
    }
}
```

### 1.4 列表的MVVM设计

```swift
// 定义模型
struct Product {
    let id: String
    let title: String
    let subtitle: String
    let price: Double
    let originalPrice: Double
    let images: [String]
    let comments: [Comment]

    let stock: Int

    let coupons: [Coupon]
    let detailHTML: String
    let recommendProducts: [Product]
    ...

/// ViewModel，参考下面的（这里的主要做解释）
class ProductDetailVM: ObservableObject {
    @Published var headerState // 顶部区域数据
    @Published var priceState // 价格区域数据
    @Published var commentState // 评论区数据
    @Published var bottomState // 底部数据
}

/// 顶部区域数据作为一个state来管理
struct HeaderState:Identifiable { // 这里的state只保留需要的数据
    let title: String
    let subtitle: String
    let images: [String]
}

/// 价格区域数据作为一个state来管理
struct PriceState:Identifiable{
    let priceText: String
    let originalPriceText: String
    let couponText: String?
}
/// 评论区域数据作为一个state来管理
struct CommentState:Identifiable {
    let comments: [CommentCellState]
}

// ⚠️在View层，尽量小的颗粒去使用数据
Text(vm.product.xxx) // ❌
PriceView( // ✅
    price: vm.priceText
)

// 其中Cell的注意点
// 尽量纯数据，不要整个 vm 传进去
ProductCell(
    title: xxx,
    price: xxx,
    image: xxx
)

//ViewModel
class ProductListVM: ObservableObject {
    @Published var items: [ProductCellState] = []
    @Published var favoriteIDs: Set<String> = [] // 收藏状态，cell上数值变化和不变化的分开处理
    private var products: [Product] = []
    func load() {
        let result: [Product] = mockProducts()
        products = result
        items = result.map {
            ProductCellState(
                id: $0.id,
                title: $0.title,
                priceText: "¥\($0.price)",
                salesText: "已售\($0.sales)",
                imageURL: $0.imageURL
            )
        }
    }
   	func toggleFavorite(id: String) {
        if favoriteIDs.contains(id) {
            favoriteIDs.remove(id)
        } else {
            favoriteIDs.insert(id)
        }
    }
}

struct ProductListView: View {
    @StateObject var vm = ProductListVM()、
 		
    var body: some View {
        List(vm.items) { item in
                        
            // 这里用的是闭包的方式
            ProductCell(item: item，isFavorite: vm.favoriteIDs.contains(item.id),
                onFavoriteTap: {
                    vm.toggleFavorite(id: item.id)
                })
            // 另外一种用Bind的方式
            ProductCell(item: item,
                        isFavorite: Binding(
                            get: {
                                vm.favoriteIDs.contains(item.id)
                            },
                            set: { newValue in
                                vm.setFavorite(item.id, newValue: newValue)
                            }
    )
)
                        
        }
        .task {
            vm.load()
        }
    }
}

struct ProductCell: View {
    let item: ProductCellState
    let isFavorite: Bool // 是否点赞
    // 这里用的是闭包的方式
  	let onFavoriteTap: () -> Void
  	// 这里用的是Bind的方式
  	@Binding var isFavorite: Bool
    var body: some View {
        HStack {
            AsyncImage(url: URL(string: item.imageURL))
            VStack(alignment: .leading) {
                Text(item.title)
                Text(item.priceText)
                Text(item.salesText)
            }
            Button {
              // 这里用的是闭包的方式
                onFavoriteTap()
              // 这里用的是Bind的方式
              isFavorite.toggle()
            } label: {
                Image(systemName: isFavorite ? "heart.fill" : "heart")
            }
        }
    }
}
```

## EnvironmentObject

全局依赖注入

```swift
/// 全局的一个对象
class AppState: ObservableObject {
    @Published var isLogin = false
}

/// 在某处注入
.environmentObject(AppState())

/// 在某处使用全局环境变量
@EnvironmentObject var appState: AppState
```

```swift
class UserVM: ObservableObject {

    @Published var name = ""

    @Published var age = 0
}

/// iOS17 引入来 Observation： @Observable。甚至不需要 Published
@Observable 
class UserVM {
    var name = ""
    var age = 0
```
